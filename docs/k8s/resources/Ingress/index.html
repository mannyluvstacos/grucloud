Ingress

Provides a Kubernetes Ingress

Examples#

Create a Ingress for minikube#

exports.createIngress = async ({
provider,
config: { restServer, ui },
resources: { namespace, serviceWebServer, serviceRestServer },
}) => {
return provider.makeIngress({
name: "ingress",
dependencies: {
namespace,
serviceWebServer,
serviceRestServer,
},
properties: () => ({
metadata: {
annotations: {
"nginx.ingress.kubernetes.io/use-regex": "true",
},
},
spec: {
rules: [
{
http: {
paths: [
{
path: "/api/.*",
pathType: "Prefix",
backend: {
service: {
name: restServer.serviceName,
port: {
number: restServer.port,
},
},
},
},
],
},
},
{
http: {
paths: [
{
path: "/.*",
pathType: "Prefix",
backend: {
service: {
name: ui.serviceName,
port: {
number: ui.port,
},
},
},
},
],
},
},
],
},
}),
});
